from django.contrib import admin from django.urls import path from django.template.response import TemplateResponse from .models import Candidate, Interview from .services import ( get_kpis, get_status_distribution, get_conversion_rates, get_stuck_candidates, ) class GoorrAdminSite(admin.AdminSite): site_header = "Goorr Analytics Dashboard" def get_urls(self): urls = super().get_urls() custom_urls = [ path("", self.admin_view(self.dashboard_view), name="dashboard"), ] return custom_urls + urls def dashboard_view(self, request): context = dict( self.each_context(request), kpis=get_kpis(), status_data=get_status_distribution(), conversions=get_conversion_rates(), stuck_candidates=get_stuck_candidates(), ) return TemplateResponse(request, "admin/dashboard.html", context) admin_site = GoorrAdminSite(name="goorr_admin") admin_site.register(Candidate) admin_site.register(Interview)
